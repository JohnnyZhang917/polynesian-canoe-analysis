model
{ 
   for( t in 1 : T ) {
      alpha[t] ~ dnorm(0, .1)
   }   
   beta ~ dnorm(0, .01)
   kappa3 ~ dnorm(0, .01)   
   kappa4 ~ dnorm(0, .01)   
   lambda ~ dnorm(0, .01)
  for( t in 1 : T ) {	for( i in 1:I ){		
		logit(p[t,i]) <- alpha[t] + beta * a[i] + kappa3 * nrh[i] + kappa4 * nrl[i] + lambda * sphere[t,i] 		
		}
	}	
   for( t in 1 : T ) {
      for( i in 1 : I ) {
         Y[t , i] ~ dbin(p[t , i], 1)
      }
   }
}
