model{ 
    for( t in 1 : T ) {
        alpha[t] ~ dnorm(0, .1)
    }
    for( t in 1 : T ) {
        for( i in 1 : I ) {
            logit(p[t , i]) <- alpha[t]
        }
    }
    for( t in 1 : T ) {
        for( i in 1 : I ) {
            Y[t , i] ~ dbin(p[t , i], 1)
        }
    }
}
